### 单链表

```cpp
struct node {
    int val,next;
};
vector<node> p(N);
int idx;
//初始化，p[0]代表头节点
void init(){
    p[0].next = - 1;
    idx = 1;
}
//查找x返回下标，找不到返回-1
int find (int x) {
    int now = p[0].next;
    while(p[now].val == x) return now;
    return -1;
}
//头插x
void inserthead(int x) {
    p[idx].next = p[0].next,p[idx].val = x;
    p[0].next = idx ++;
}
//删除下标为k + 1的节点
void del(int k) {
    if(k == 0) {p[k].next = p[p[k].next].next; return ;}
    p[k].next = p[p[k].next].next;
}
//在下标为k的节点后面插入一个x
void insert(int k, int x) {
    p[idx].next = p[k].next,p[idx].val = x;
    p[k].next = idx ++;
}
```

### 双链表

```cpp
struct node {
    int val,per,next;
}p[N];
int idx;
//初始化
void init() {
    p[0].next = 1;
    p[1].per = 0;
    idx = 2;
}
//插入
void insert(int k, int x) {
    p[idx].val = x, p[idx].per = k,p[idx].next = p[k].next;
    p[p[k].next].per = idx,p[k].next = idx ++;
}
//c
void del (int k) {
    p[p[k].per].next = p[k].next, p[p[k].next].per = p[k].per;
}
```

### 栈

```cpp

hh = tt = 0;
int S[N];
void insert(int x) {
	S[tt ++] = x;
}

	tt --;
}

    return S[tt];
}
```

### 单调栈

```cpp
int hh, tt;
int S[N];
void pop() {
	tt --;
}
void push(int x) {
    while(S[tt] >= x && tt > hh) pop();
	S[tt ++] = x;
}
int top() {
    return S[tt];
}
```

### 字典树(Trie)

```cpp
int son[N][M], cnt[N], idx;
char str[N];
void insert(char *str) {
    int p = 0;
    for(int i = 0;str[i];i ++) {
        int u = str[i] - 'a';
        if(!son[p][u]) son[p][u] = ++ idx;
        p = son[p][u];
    }
    cnt[p] ++;
}
int query(char *str) {
    int p = 0;
    for(int i = 0;str[i];i ++) {
		int u = str[i] - 'a';
        if(!son[p][u]) return 0;
        p = son[p][u];
    }
    return cnt[p];
}
```

### 并查集

```cpp
int p[N];
int find(int x) {
    if(p[x] != x) p[x] = find(p[x]);//路径压缩优化
    return p[x];
}
void merge(int x, int y) {
    if(find(x) != find(y)) p[find(x)] = find(y);
    return ;
}
//启发式合并：每个根节点维护一个size()表示集合当中的数量，将size()较小的祖宗做size()较大的祖宗的儿子
//按秩合并：每一个根节点维护一个hig()表示集合中树的高度，将hig()较小的祖宗做hig()较大的祖宗的儿子
//可以比较log(size())的启发式合并来达到按秩合并的效果
```

### 带权并查集

在并查集的过程中额外维护一些信息，比如说集合中每两个点之间的关系；集合的整体属性等；

前者一般在find函数的实现更新权值，后者一般在merge函数中更新前权值。

### 堆

//用一维数组来模拟堆，根节点的的下标为1，对每一个节点的左儿子的下标为2*x，右儿子为2*x + 1
//堆的可以实现的操作：1、插入一个数；2、查询最小（大）的元素；3、删除最小（大）的元素；4、修改任一元素；5、合并两个堆
//核心操作：up和down两个函数
int h[N];
void up(int x) {
    while (x > 1 && h[x] < h[x / 2]) {
        swap(h[x], h[x / 2]);
        x /= 2;
    }
}
void down(int x) {
    while (x * 2 <= n) {
        int t = x * 2;
        if (t + 1 <= n && h[t + 1] < h[t]) t++;
        if (h[t] >= h[x]) break;
        swap(h[x], h[t]);
        x = t;
    }
}
//一、建堆
for(int i = n >> 1;i;i --) {
	down(i);
}
//二、插入一个元素
h[++n] = x;
up(n);
//三、查询最小（大）的元素
h[1];
//四、删除最小（大）的元素
